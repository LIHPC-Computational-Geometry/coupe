11const ERROR_OTHER: c_int = -1;
12const ERROR_BAD_HEADER: c_int = -2;
13const ERROR_UNSUPPORTED_VERSION: c_int = -3;

15fn err_partition_code(err: mesh_io::partition::Error) -> c_int {
16    match err {
17        mesh_io::partition::Error::BadHeader => ERROR_BAD_HEADER,
18        mesh_io::partition::Error::UnsupportedVersion => ERROR_UNSUPPORTED_VERSION,
19        mesh_io::partition::Error::Io(_) => ERROR_OTHER,
20    }
21}

23#[unsafe(no_mangle)]
24pub unsafe extern "C" fn mio_partition_read(
25    size: *mut u64,
26    partition: *mut *mut u64,
27    fd: c_int,
28) -> c_int {
29    unsafe {
30        let f = fs::File::from_raw_fd(fd);
31        let r = io::BufReader::new(f);
32        match mesh_io::partition::read(r) {
33            Ok(p) => {
34                let mut p = p.into_boxed_slice();
35                *size = p.len() as u64;
36                *partition = p.as_mut_ptr() as *mut u64;
37                0
38            }
39            Err(err) => err_partition_code(err),
40        }
41    }
42}

44#[unsafe(no_mangle)]
45pub unsafe extern "C" fn mio_partition_write(fd: c_int, size: u64, partition: *const u64) -> c_int {
46    unsafe {
47        let f = fs::File::from_raw_fd(fd);
48        let mut w = io::BufWriter::new(f);
49        let p = std::slice::from_raw_parts(partition, size as usize)
50            .iter()
51            .map(|id| *id as usize);
52        if mesh_io::partition::write(&mut w, p).is_err() {
53            return ERROR_OTHER;
54        }
55        match w.into_inner() {
56            Ok(f) => {
57                let _ = f.into_raw_fd(); // raw file descriptor is fd
58                0
59            }
60            Err(_) => ERROR_OTHER,
61        }
62    }
63}

75    unsafe {
76        let f = fs::File::from_raw_fd(fd);
77        let mut r = io::BufReader::new(f);

100#[unsafe(no_mangle)]
101pub unsafe extern "C" fn mio_weights_first_criterion(
102    criterion: *mut f64,
103    weights: *const mesh_io::weight::Array,
104) {
105    unsafe {
106        assert!(!weights.is_null());
107
108        let weights = &*weights;
109        let len = match weights {
110            mesh_io::weight::Array::Integers(is) => is.len(),
111            mesh_io::weight::Array::Floats(fs) => fs.len(),
112        };
113        let criterion = std::slice::from_raw_parts_mut(criterion, len);
114
115        match weights {
116            mesh_io::weight::Array::Integers(is) => {
117                for (c, w) in criterion.iter_mut().zip(is) {
118                    *c = w[0] as f64;
119                }
120            }
121            mesh_io::weight::Array::Floats(fs) => {
122                for (c, w) in criterion.iter_mut().zip(fs) {
123                    *c = w[0];
124                }
125            }
126        }
127    }
128}

206    element: *mut MeditElement,

209) {

212        if let Some((el_type, el_nodes, _el_ref)) = (*medit).elements().nth(element_idx) {
