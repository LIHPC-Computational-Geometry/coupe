31unsafe fn array_assume_init<T, const N: usize>(array: [MaybeUninit<T>; N]) -> [T; N] {
32    // SAFETY:
33    // * The caller guarantees that all elements of the array are initialized
34    // * `MaybeUninit<T>` and T are guaranteed to have the same layout
35    // * `MaybeUninit` does not drop, so there are no double-frees
36    // And thus the conversion is safe
37    //std::intrinsics::assert_inhabited::<[T; N]>();
38    unsafe { (&array as *const _ as *const [T; N]).read() }
39}

41fn array_init<F, T, const N: usize>(mut f: F) -> [T; N]
42where
43    F: FnMut(usize) -> T,
44{
45    unsafe {
46        let mut output: [MaybeUninit<T>; N] = MaybeUninit::uninit().assume_init();
47        for (i, u) in output.iter_mut().enumerate() {
48            u.write(f(i));
49        }
50        array_assume_init(output)
51    }
52}

54fn array_map_mut<'a, F, T, U, const N: usize>(input: &'a mut [T; N], mut map: F) -> [U; N]
55where
56    F: FnMut(&'a mut T) -> U,
57    U: 'a,
58{
59    unsafe {
60        let mut output: [MaybeUninit<U>; N] = MaybeUninit::uninit().assume_init();
61        for (u, t) in output.iter_mut().zip(input) {
62            u.write(map(t));
63        }
64        array_assume_init(output)
65    }
66}

68fn array_unzip<F, T, U, V, const N: usize>(input: [T; N], mut map: F) -> ([U; N], [V; N])
69where
70    F: FnMut(T) -> (U, V),
71{
72    unsafe {
73        let mut output1: [MaybeUninit<U>; N] = MaybeUninit::uninit().assume_init();
74        let mut output2: [MaybeUninit<V>; N] = MaybeUninit::uninit().assume_init();
75        for ((u, v), t) in output1.iter_mut().zip(&mut output2).zip(input) {
76            let (new_u, new_v) = map(t);
77            u.write(new_u);
78            v.write(new_v);
79        }
80        (array_assume_init(output1), array_assume_init(output2))
81    }
82}

105struct Items<'a, const D: usize, W> {
106    points: [&'a mut [f32]; D],
107    weights: &'a mut [W],
108    parts: &'a mut [&'a AtomicUsize],

111/// Return value of [rcb_split] and [par_rcb_split].
112struct SplitResult<'a, const D: usize, W> {
113    left: Items<'a, D, W>,
114    right: Items<'a, D, W>,
115    /// Weight of the left part, used to compute the sum for the next iteration.
116    weight_left: W,
117    /// Coordinate value of the split, used to compute the [BoundingBox]es.
118    split_pos: f32,
119}

121/// Scalar version of [reorder_split].
122fn reorder_split_scalar<const D: usize, W>(
123    mut items: Items<'_, D, W>,
124    pivot: usize,
125    coord: usize,
126) -> (Items<'_, D, W>, Items<'_, D, W>) {
127    for ps in &mut items.points {
128        ps.swap(0, pivot);
129    }
130    items.weights.swap(0, pivot);
131    items.parts.swap(0, pivot);
132
133    let (pivot, mut points) = array_unzip_mut(&mut items.points, |p| p.split_at_mut(1));
134    let pivot = pivot[coord][0];
135    let (_, weights) = items.weights.split_at_mut(1);
136    let (_, parts) = items.parts.split_at_mut(1);
137
138    let mut l = 0;
139    let mut r = points[0].len();
140    loop {
141        unsafe {
142            let coords = points.get_unchecked(coord);
143            while l < r && *coords.get_unchecked(l) < pivot {
144                l += 1;
145            }
146            while l < r && pivot <= *coords.get_unchecked(r - 1) {
147                r -= 1;
148            }
149            if r <= l {
150                break;
151            }
152            r -= 1;
153            for p in &mut points {
154                p.swap(l, r);
155            }
156            weights.swap(l, r);
157            parts.swap(l, r);
158            l += 1;
159        }
160    }
161
162    for p in &mut items.points {
163        p.swap(0, l);
164    }
165    items.weights.swap(0, l);
166    items.parts.swap(0, l);
167    let (points_left, points_right) = array_unzip(items.points, |p| p.split_at_mut(l));
168    let (weights_left, weights_right) = items.weights.split_at_mut(l);
169    let (parts_left, parts_right) = items.parts.split_at_mut(l);
170    let left = Items {
171        points: points_left,
172        weights: weights_left,
173        parts: parts_left,
174    };
175    let right = Items {
176        points: points_right,
177        weights: weights_right,
178        parts: parts_right,
179    };
180    (left, right)
181}

186/// AVX512 (VL+F) version of [reorder_split].
187#[cfg(feature = "avx512")]
188fn reorder_split_avx512<const D: usize, W>(
189    items: Items<'_, D, W>,
190    pivot: usize,
191    coord: usize,
192) -> (Items<'_, D, W>, Items<'_, D, W>) {
193    use std::arch::x86_64::__m256;
194    use std::arch::x86_64::__m512i;
195    use std::arch::x86_64::__mmask8;
196    use std::arch::x86_64::_CMP_LT_OQ;
197    use std::arch::x86_64::_mm256_cmp_ps_mask;
198    use std::arch::x86_64::_mm256_loadu_ps;
199    use std::arch::x86_64::_mm256_mask_compressstoreu_ps;
200    use std::arch::x86_64::_mm256_set1_ps;
201    use std::arch::x86_64::_mm512_loadu_epi64;
202    use std::arch::x86_64::_mm512_mask_compressstoreu_epi64;
203
204    let pivot = items.points[coord][pivot];
205
206    unsafe {
207        let pivot_val: __m256 = _mm256_set1_ps(pivot);
208
209        let mut lw = 0;
210        let mut l = AVX2_REGISTER_BYTES;
211        let mut lv: [__m256; D] = array_init(|coord| _mm256_loadu_ps(items.points[coord].as_ptr()));
212        let mut lv_weights = _mm512_loadu_epi64(items.weights.as_ptr() as *const i64);
213        let mut lv_parts = _mm512_loadu_epi64(items.parts.as_ptr() as *const i64);
214
215        let mut rw = items.points[coord].len();
216        let mut r = items.points[coord].len() - AVX2_REGISTER_BYTES;
217        let mut rv: [__m256; D] =
218            array_init(|coord| _mm256_loadu_ps(items.points[coord].as_ptr().add(r)));
219        let mut rv_weights = _mm512_loadu_epi64(items.weights.as_ptr().add(r) as *const i64);
220        let mut rv_parts = _mm512_loadu_epi64(items.parts.as_ptr().add(r) as *const i64);
221
222        while l + AVX2_REGISTER_BYTES <= r {
223            let free_left = l - lw;
224            let free_right = rw - r;
225
226            let val: [__m256; D];
227            let val_weights: __m512i;
228            let val_parts: __m512i;
229            if free_left <= free_right {
230                val = lv;
231                val_weights = lv_weights;
232                val_parts = lv_parts;
233                lv = array_init(|coord| _mm256_loadu_ps(items.points[coord].as_ptr().add(l)));
234                lv_weights = _mm512_loadu_epi64(items.weights.as_ptr().add(l) as *const i64);
235                lv_parts = _mm512_loadu_epi64(items.parts.as_ptr().add(l) as *const i64);
236                l += AVX2_REGISTER_BYTES;
237            } else {
238                val = rv;
239                val_weights = rv_weights;
240                val_parts = rv_parts;
241                r -= AVX2_REGISTER_BYTES;
242                rv = array_init(|coord| _mm256_loadu_ps(items.points[coord].as_ptr().add(r)));
243                rv_weights = _mm512_loadu_epi64(items.weights.as_ptr().add(r) as *const i64);
244                rv_parts = _mm512_loadu_epi64(items.parts.as_ptr().add(r) as *const i64);
245            }
246
247            let mask: __mmask8 = _mm256_cmp_ps_mask::<_CMP_LT_OQ>(val[coord], pivot_val);
248
249            let nb_low = mask.count_ones() as usize;
250            let nb_high = AVX2_REGISTER_BYTES - nb_low;
251
252            for coord in 0..D {
253                _mm256_mask_compressstoreu_ps(
254                    items.points[coord].as_mut_ptr().add(lw),
255                    mask,
256                    val[coord],
257                );
258            }
259            _mm512_mask_compressstoreu_epi64(
260                items.weights.as_mut_ptr().add(lw) as *mut i64, // TODO: test
261                mask,
262                val_weights,
263            );
264            _mm512_mask_compressstoreu_epi64(
265                items.parts.as_mut_ptr().add(lw) as *mut i64, // TODO: test
266                mask,
267                val_parts,
268            );
269            lw += nb_low;
270
271            rw -= nb_high;
272            for coord in 0..D {
273                _mm256_mask_compressstoreu_ps(
274                    items.points[coord].as_mut_ptr().add(rw),
275                    !mask,
276                    val[coord],
277                );
278            }
279            _mm512_mask_compressstoreu_epi64(
280                items.weights.as_mut_ptr().add(rw) as *mut i64, // TODO: test
281                !mask,
282                val_weights,
283            );
284            _mm512_mask_compressstoreu_epi64(
285                items.parts.as_mut_ptr().add(rw) as *mut i64, // TODO: test
286                !mask,
287                val_parts,
288            );
289        }
290
291        let remaining = r - l;
292        let val: [__m256; D] =
293            array_init(|coord| _mm256_loadu_ps(items.points[coord].as_ptr().add(l)));
294        let val_weights = _mm512_loadu_epi64(items.weights.as_ptr().add(l) as *const i64);
295        let val_parts = _mm512_loadu_epi64(items.parts.as_ptr().add(l) as *const i64);
296        let mask: __mmask8 = _mm256_cmp_ps_mask::<_CMP_LT_OQ>(val[coord], pivot_val);
297        let mask_low = mask & !(0xff << remaining);
298        let mask_high = !mask & !(0xff << remaining);
299        let nb_low = mask_low.count_ones() as usize;
300        let nb_high = mask_high.count_ones() as usize;
301        for coord in 0..D {
302            _mm256_mask_compressstoreu_ps(
303                items.points[coord].as_mut_ptr().add(lw),
304                mask_low,
305                val[coord],
306            );
307        }
308        _mm512_mask_compressstoreu_epi64(
309            items.weights.as_mut_ptr().add(lw) as *mut i64, // TODO: test
310            mask_low,
311            val_weights,
312        );
313        _mm512_mask_compressstoreu_epi64(
314            items.parts.as_mut_ptr().add(lw) as *mut i64, // TODO: test
315            mask_low,
316            val_parts,
317        );
318        lw += nb_low;
319        rw -= nb_high;
320        for coord in 0..D {
321            _mm256_mask_compressstoreu_ps(
322                items.points[coord].as_mut_ptr().add(rw),
323                mask_high,
324                val[coord],
325            );
326        }
327        _mm512_mask_compressstoreu_epi64(
328            items.weights.as_mut_ptr().add(rw) as *mut i64, // TODO: test
329            mask_high,
330            val_weights,
331        );
332        _mm512_mask_compressstoreu_epi64(
333            items.parts.as_mut_ptr().add(rw) as *mut i64, // TODO: test
334            mask_high,
335            val_parts,
336        );
337
338        let mask: __mmask8 = _mm256_cmp_ps_mask::<_CMP_LT_OQ>(lv[coord], pivot_val);
339        let nb_low = mask.count_ones() as usize;
340        let nb_high = AVX2_REGISTER_BYTES - nb_low;
341        for coord in 0..D {
342            _mm256_mask_compressstoreu_ps(
343                items.points[coord].as_mut_ptr().add(lw),
344                mask,
345                lv[coord],
346            );
347        }
348        _mm512_mask_compressstoreu_epi64(
349            items.weights.as_mut_ptr().add(lw) as *mut i64, // TODO: test
350            mask,
351            lv_weights,
352        );
353        _mm512_mask_compressstoreu_epi64(
354            items.parts.as_mut_ptr().add(lw) as *mut i64, // TODO: test
355            mask,
356            lv_parts,
357        );
358        lw += nb_low;
359        rw -= nb_high;
360        for coord in 0..D {
361            _mm256_mask_compressstoreu_ps(
362                items.points[coord].as_mut_ptr().add(rw),
363                !mask,
364                lv[coord],
365            );
366        }
367        _mm512_mask_compressstoreu_epi64(
368            items.weights.as_mut_ptr().add(rw) as *mut i64, // TODO: test
369            !mask,
370            lv_weights,
371        );
372        _mm512_mask_compressstoreu_epi64(
373            items.parts.as_mut_ptr().add(rw) as *mut i64, // TODO: test
374            !mask,
375            lv_parts,
376        );
377
378        let mask: __mmask8 = _mm256_cmp_ps_mask::<_CMP_LT_OQ>(rv[coord], pivot_val);
379        let nb_low = mask.count_ones() as usize;
380        let nb_high = AVX2_REGISTER_BYTES - nb_low;
381        for coord in 0..D {
382            _mm256_mask_compressstoreu_ps(
383                items.points[coord].as_mut_ptr().add(lw),
384                mask,
385                rv[coord],
386            );
387        }
388        _mm512_mask_compressstoreu_epi64(
389            items.weights.as_mut_ptr().add(lw) as *mut i64, // TODO: test
390            mask,
391            rv_weights,
392        );
393        _mm512_mask_compressstoreu_epi64(
394            items.parts.as_mut_ptr().add(lw) as *mut i64, // TODO: test
395            mask,
396            rv_parts,
397        );
398        lw += nb_low;
399        rw -= nb_high;
400        for coord in 0..D {
401            _mm256_mask_compressstoreu_ps(
402                items.points[coord].as_mut_ptr().add(rw),
403                !mask,
404                rv[coord],
405            );
406        }
407        _mm512_mask_compressstoreu_epi64(
408            items.weights.as_mut_ptr().add(rw) as *mut i64, // TODO: test
409            !mask,
410            rv_weights,
411        );
412        _mm512_mask_compressstoreu_epi64(
413            items.parts.as_mut_ptr().add(rw) as *mut i64, // TODO: test
414            !mask,
415            rv_parts,
416        );
417
418        let (points_left, points_right) = array_unzip(items.points, |p| p.split_at_mut(lw));
419        let (weights_left, weights_right) = items.weights.split_at_mut(lw);
420        let (parts_left, parts_right) = items.parts.split_at_mut(lw);
421        let left = Items {
422            points: points_left,
423            weights: weights_left,
424            parts: parts_left,
425        };

427            points: points_right,

430        };

432    }

435/// Split the arrays in `items` into the ones that have `points[i][coord]`
436/// strictly lower than `items.point[pivot][coord]` and the others.
437fn reorder_split<const D: usize, W>(
438    items: Items<'_, D, W>,
439    pivot: usize,
440    coord: usize,
441) -> (Items<'_, D, W>, Items<'_, D, W>) {
442    #[cfg(feature = "avx512")]
443    if is_x86_feature_detected!("avx512f")
444        && is_x86_feature_detected!("avx512vl")
445        && std::mem::size_of::<W>() == 8
446        && std::mem::size_of::<&AtomicUsize>() == 8
447        && items.parts.len() >= 2 * AVX2_REGISTER_BYTES
448    {
449        return reorder_split_avx512(items, pivot, coord);
450    }
451
452    reorder_split_scalar(items, pivot, coord)
453}

455/// Splits the given items into two sets of similar weights (parallel version).
456fn par_rcb_split<const D: usize, W>(
457    items: Items<'_, D, W>,
458    coord: usize,
459    tolerance: f64,
460    mut min: f32,
461    mut max: f32,
462    sum: W,
463) -> SplitResult<'_, D, W>
464where
465    W: RcbWeight,
466{
467    let span = tracing::info_span!("par_rcb_split");
468    let _enter = span.enter();
469
470    let mut prev_count_left = usize::MAX;
471    loop {
472        let split_target = (min + max) / 2.0;
473
474        // count_left: the number of points that are on the left of split_target
475        // weight_left: the weight of all those points
476        // nearest_idx: the index in `items` of the point that is 1/ on the
477        //    right side of split_target and 2/ the nearest to split_target
478        let (count_left, weight_left, nearest_idx, nearest_distance) = items.points[coord]
479            .par_iter()
480            .with_min_len(4096)
481            .zip(&*items.weights)
482            .enumerate()
483            .fold(
484                || (0, W::default(), None, f32::INFINITY),
485                |(count, weight_left, mut nearest_idx, mut nearest_distance),
486                 (idx, (point, weight))| {
487                    let distance = point - split_target;
488                    if distance < 0.0 {
489                        (
490                            count + 1,
491                            weight_left + *weight,
492                            nearest_idx,
493                            nearest_distance,
494                        )
495                    } else {
496                        if distance < nearest_distance {
497                            nearest_distance = distance;
498                            nearest_idx = Some(idx);
499                        }
500                        (count, weight_left, nearest_idx, nearest_distance)
501                    }
502                },
503            )
504            .reduce(
505                || (0, W::default(), None, f32::INFINITY),
506                |(count0, weight0, nearest_idx0, nearest_distance0),
507                 (count1, weight1, nearest_idx1, nearest_distance1)| {
508                    let (nearest_idx, nearest_distance) = if nearest_distance0 < nearest_distance1 {
509                        (nearest_idx0, nearest_distance0)
510                    } else {
511                        (nearest_idx1, nearest_distance1)
512                    };
513                    (
514                        count0 + count1,
515                        weight0 + weight1,
516                        nearest_idx,
517                        nearest_distance,
518                    )
519                },
520            );
521
522        let nearest_idx = match nearest_idx {
523            Some(v) => v,
524            // Both following cases happen when all points are of the left side
525            // of the split_target.  This is the case when min and max are set
526            // too loosely, so we let it happen once. If this happens twice,
527            // then `prev_count_left` will be the equal to `count_left`.
528            None if prev_count_left == count_left => {
529                return SplitResult {
530                    left: items,
531                    right: Items {
532                        points: array_init(|_| &mut [][..]),
533                        weights: &mut [],
534                        parts: &mut [],
535                    },
536                    weight_left: sum,
537                    split_pos: max,
538                };
539            }
540            None => {
541                max = split_target;
542                prev_count_left = count_left;
543                continue;
544            }
545        };
546
547        let imbalance = {
548            let ideal_weight_left = sum.to_f64().unwrap() / 2.0;
549            let weight_left = weight_left.to_f64().unwrap();
550            f64::abs((weight_left - ideal_weight_left) / ideal_weight_left)
551        };
552        if count_left == prev_count_left // there is no point between min and max
553            || max <= split_target + nearest_distance // or between split_target and max
554            || imbalance <= tolerance
555        {
556            let (left, right) = reorder_split(items, nearest_idx, coord);
557            return SplitResult {
558                left,
559                right,
560                weight_left,
561                split_pos: split_target,
562            };
563        }
564        prev_count_left = count_left;
565
566        let weight_right = sum - weight_left;
567        if weight_left < weight_right {
568            min = split_target;
569        } else {
570            max = split_target;
571        }
572    }
573}

575fn rcb_recurse<const D: usize, W>(
576    items: Items<'_, D, W>,
577    iter_count: usize,
578    iter_id: usize,
579    coord: usize,
580    tolerance: f64,
581    sum: W,
582    bb: BoundingBox<D>,
583) where
584    W: RcbWeight,
585{
586    if items.parts.is_empty() {
587        return;
588    }
589    if iter_count == 0 {
590        let span = tracing::info_span!(
591            "rcb_recurse: apply_part_id",
592            item_count = items.parts.len(),
593            iter_id,
594        );
595        let _enter = span.enter();
596
597        items
598            .parts
599            .into_par_iter()
600            .for_each(|part| part.store(iter_id, Ordering::Relaxed));
601        return;
602    }
603
604    let min = bb.p_min[coord] as f32;
605    let max = bb.p_max[coord] as f32;
606    let SplitResult {
607        left,
608        right,
609        weight_left,
610        split_pos,
611    } = par_rcb_split(items, coord, tolerance, min, max, sum);
612
613    let mut bb_left = bb.clone();
614    bb_left.p_max[coord] = split_pos as f64;
615    let mut bb_right = bb;
616    bb_right.p_min[coord] = split_pos as f64;
617
618    rayon::join(
619        || {
620            rcb_recurse(
621                left,
622                iter_count - 1,
623                2 * iter_id + 1,
624                (coord + 1) % D,
625                tolerance,
626                weight_left,
627                bb_left,
628            )
629        },
630        || {
631            rcb_recurse(
632                right,
633                iter_count - 1,
634                2 * iter_id + 2,
635                (coord + 1) % D,
636                tolerance,
637                sum - weight_left,
638                bb_right,
639            )
640        },
641    );
642}

644fn rcb<const D: usize, P, W>(
645    partition: &mut [usize],
646    points: P,
647    weights: W,
648    iter_count: usize,
649    tolerance: f64,
650) -> Result<(), Error>
651where
652    P: rayon::iter::IntoParallelIterator<Item = PointND<D>>,
653    P::Iter: rayon::iter::IndexedParallelIterator + Clone,
654    W: rayon::iter::IntoParallelIterator,
655    W::Item: RcbWeight,
656    W::Iter: rayon::iter::IndexedParallelIterator,
657{
658    let points = points.into_par_iter();
659    let weights = weights.into_par_iter();
660
661    if weights.len() != partition.len() {
662        return Err(Error::InputLenMismatch {
663            expected: partition.len(),
664            actual: weights.len(),
665        });
666    }
667    if points.len() != partition.len() {
668        return Err(Error::InputLenMismatch {
669            expected: partition.len(),
670            actual: points.len(),
671        });
672    }
673
674    let mut coords = array_init(|coord| {
675        points
676            .clone()
677            .map(|point| point[coord] as f32)
678            .collect::<Vec<f32>>()
679    });
680    let mut weights: Vec<_> = weights.collect();
681
682    let atomic_partition = crate::as_atomic(partition);
683    let mut atomic_partition: Vec<&AtomicUsize> = atomic_partition.par_iter().collect();
684    let sum = weights.par_iter().cloned().sum();
685    let bb = match BoundingBox::from_points(points) {
686        Some(v) => v,
687        None => return Ok(()), // `items` is empty.
688    };
689
690    let points = array_map_mut(&mut coords, |coord| &mut coord[..]);
691    let items = Items {
692        points,
693        weights: &mut weights,
694        parts: &mut atomic_partition,
695    };
696    rcb_recurse(items, iter_count, 0, 0, tolerance, sum, bb);
697
698    // Part IDs must start from zero.
699    let part_id_offset = *partition.par_iter().min().unwrap();
700    partition
701        .par_iter_mut()
702        .for_each(|part_id| *part_id -= part_id_offset);
703
704    Ok(())
705}

724/// # Recursive Coordinate Bisection algorithm
725///
726/// Partitions a mesh based on the nodes coordinates and coresponding weights.
727///
728/// This is the most simple and straightforward geometric algorithm. It operates
729/// as follows for a N-dimensional set of points:
730///
731/// At each iteration, select a vector `n` of the canonical basis
732/// `(e_0, ..., e_{n-1})`. Then, split the set of points with an hyperplane
733/// orthogonal to `n`, such that the two parts of the splits are evenly
734/// weighted. Finally, recurse by reapplying the algorithm to the two parts with
735/// an other normal vector selection.
736///
737/// # Example
738///
739/// ```rust
740/// # fn main() -> Result<(), coupe::Error> {
741/// use coupe::Partition as _;
742/// use coupe::Point2D;
743///
744/// let points = [
745///     Point2D::new(1., 1.),
746///     Point2D::new(-1., 1.),
747///     Point2D::new(1., -1.),
748///     Point2D::new(-1., -1.),
749/// ];
750/// let weights = [1; 4];
751/// let mut partition = [0; 4];
752///
753/// // Generate a partition of 4 parts (2 splits).
754/// coupe::Rcb { iter_count: 2, ..Default::default() }
755///     .partition(&mut partition, (points, weights))?;
756///
757/// // All points are in different parts.
758/// for i in 0..4 {
759///     for j in 0..4 {
760///         if j == i {
761///             continue
762///         }
763///         assert_ne!(partition[i], partition[j])
764///     }
765/// }
766/// # Ok(())
767/// # }
768/// ```
769///
770/// # References
771///
772/// Berger, M. J. and Bokhari, S. H., 1987. A partitioning strategy for
773/// nonuniform problems on multiprocessors. *IEEE Transactions on Computers*,
774/// C-36(5):570–580. <doi:10.1109/TC.1987.1676942>.
775///
776/// Bramas, B., 2017. A Novel Hybrid Quicksort Algorithm Vectorized using
777/// AVX-512 on Intel Skylake. *International Journal of Advanced Computer
778/// Science and Applications*, 8(10). <doi:10.14569/IJACSA.2017.081044>.
779#[derive(Clone, Copy, Debug, Default)]
780pub struct Rcb {
781    /// The number of iterations of the algorithm. This will yield a partition
782    /// of at most `2^num_iter` parts.
783    ///
784    /// If this equals zero, RCB will not do anything.
785    pub iter_count: usize,
786
787    /// Tolerance on the normalized imbalance, for each split.  Please note that
788    /// the overall imbalance might end up above this threshold.
789    ///
790    /// Negative values are interpreted as zeroes.
791    pub tolerance: f64,
792}
793
794impl<const D: usize, P, W> crate::Partition<(P, W)> for Rcb
795where
796    P: rayon::iter::IntoParallelIterator<Item = PointND<D>>,
797    P::Iter: rayon::iter::IndexedParallelIterator + Clone,
798    W: rayon::iter::IntoParallelIterator,
799    W::Item: RcbWeight,
800    W::Iter: rayon::iter::IndexedParallelIterator,
801{
802    type Metadata = ();
803    type Error = Error;
804
805    fn partition(
806        &mut self,
807        part_ids: &mut [usize],
808        (points, weights): (P, W),
809    ) -> Result<Self::Metadata, Self::Error> {
810        rcb(part_ids, points, weights, self.iter_count, self.tolerance)
811    }
812}

814// pub because it is also useful for multijagged and required for benchmarks
815pub fn axis_sort<const D: usize>(
816    points: &[PointND<D>],
817    permutation: &mut [usize],
818    current_coord: usize,
819) {
820    permutation.par_sort_unstable_by(|i1, i2| {
821        if points[*i1][current_coord] < points[*i2][current_coord] {
822            cmp::Ordering::Less
823        } else {
824            cmp::Ordering::Greater
825        }
826    })
827}

829fn rib<const D: usize, W>(
830    partition: &mut [usize],
831    points: &[PointND<D>],
832    weights: W,
833    n_iter: usize,
834    tolerance: f64,
835) -> Result<(), Error>
836where
837    Const<D>: DimSub<Const<1>>,
838    DefaultAllocator: Allocator<Const<D>, Const<D>, Buffer<f64> = ArrayStorage<f64, D, D>>
839        + Allocator<DimDiff<Const<D>, Const<1>>>,
840    W: rayon::iter::IntoParallelIterator,
841    W::Item: RcbWeight,
842    W::Iter: rayon::iter::IndexedParallelIterator,
843{
844    let obb = match OrientedBoundingBox::from_points(points) {
845        Some(v) => v,
846        None => return Ok(()),
847    };
848    let points = points.par_iter().map(|p| obb.obb_to_aabb(p));
849    // When the rotation is done, we just apply RCB
850    rcb(partition, points, weights, n_iter, tolerance)
851}

853/// # Recursive Inertial Bisection algorithm
854///
855/// Partitions a mesh based on the nodes coordinates and coresponding weights.
856///
857/// A variant of the [Recursive Coordinate Bisection algorithm][crate::Rcb]
858/// where a basis change is performed beforehand so that the first coordinate of
859/// the new basis is colinear to the inertia axis of the set of points. This has
860/// the goal of producing better shaped partition than [RCB][crate::Rcb].
861///
862/// # Example
863///
864/// ```rust
865/// # fn main() -> Result<(), coupe::Error> {
866/// use coupe::Partition as _;
867/// use coupe::Point2D;
868///
869/// // Here, the inertia axis is the y axis.
870/// // We thus expect Rib to split horizontally first.
871/// let points = [
872///     Point2D::new(1., 10.),
873///     Point2D::new(-1., 10.),
874///     Point2D::new(1., -10.),
875///     Point2D::new(-1., -10.),
876/// ];
877/// let weights = [1; 4];
878/// let mut partition = [0; 4];
879///
880/// // Generate a partition of 2 parts (1 split).
881/// coupe::Rib { iter_count: 1, ..Default::default() }
882///     .partition(&mut partition, (&points, weights))?;
883///
884/// // The two points at the top are in the same part.
885/// assert_eq!(partition[0], partition[1]);
886///
887/// // The two points at the bottom are in the same part.
888/// assert_eq!(partition[2], partition[3]);
889///
890/// // There are two different parts.
891/// assert_ne!(partition[1], partition[2]);
892/// # Ok(())
893/// # }
894/// ```
895///
896/// # Reference
897///
898/// Williams, Roy D., 1991. Performance of dynamic load balancing algorithms for
899/// unstructured mesh calculations. *Concurrency: Practice and Experience*,
900/// 3(5):457–481. <doi:10.1002/cpe.4330030502>.
901#[derive(Clone, Copy, Debug, Default)]
902pub struct Rib {
903    /// The number of iterations of the algorithm. This will yield a partition
904    /// of at most `2^num_iter` parts.
905    pub iter_count: usize,
906
907    /// Same meaning as [`Rcb::tolerance`].
908    pub tolerance: f64,
909}
910
911impl<'a, const D: usize, W> crate::Partition<(&'a [PointND<D>], W)> for Rib
912where
913    Const<D>: DimSub<Const<1>> + ToTypenum,
914    DefaultAllocator: Allocator<Const<D>, Const<D>, Buffer<f64> = ArrayStorage<f64, D, D>>
915        + Allocator<DimDiff<Const<D>, Const<1>>>,
916    W: rayon::iter::IntoParallelIterator,
917    W::Item: RcbWeight,
918    W::Iter: rayon::iter::IndexedParallelIterator,
919{
920    type Metadata = ();
921    type Error = Error;
922
923    fn partition(
924        &mut self,
925        part_ids: &mut [usize],
926        (points, weights): (&'a [PointND<D>], W),
927    ) -> Result<Self::Metadata, Self::Error> {
928        rib(part_ids, points, weights, self.iter_count, self.tolerance)
929    }
930}
